### Get progress for entire Row B
```typescript
const rowActivities = await getDocs(query(
  collection(db, 'activities'),
  where('siteId', '==', 'site123'),
  where('blockId', '==', 'block1'),
  where('row', '==', 'B')
));

// Calculate: (completed activities) / (total activities)
const progress = rowActivities.docs
  .filter(d => d.data().status === 'DONE').length / rowActivities.size * 100;
```
